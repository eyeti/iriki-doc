2 /*
3 This file uploader has a three part aim:
4 1. Put files in a yyyy/mm year-month structure so as not to max OS maximum file per directory limits.
5 2. Make files accessible via http
6 3. Make files accessible via the file/directory paths
7 
8 In essense:
9 Files exist in year/month/ directories
10 Which can then be referred to in two ways;
11 via the file system e.g /path/on/server/files/user_uploads/2017/11/avatar.png
12 via http: http://address_via_http_path/files/user_uploads/2017/11/avatar.png
13 
14 This was initially written to use php variables like DIRECTORY_SEPARATOR and __DIR__.
15 But we've come very paranoid of late (after watching the movie IT?), eschewing (great chance to use this word) variables!
16 
17 By the way, all directory paths end with '/'. Go figure!
18 
19 Authors: Isah Lata (github.com/latanoel) and Stephen Igwue(github.com/stigwue).
20 */

57 
58  $status = true;
59

61  {

63  }
64

66  }
67  else

71  }
72  }

80  public function upload($request, $wrap = true)
81  {
82  $data = $request->getData();
83 
84  //uploaded file location on server
85  $temp_path = (isset($data['file']['tmp_name']) ? $data['file']['tmp_name'] : null);
86 
87  if (is_null($temp_path) || !file_exists($temp_path))
88  {
89  return \iriki\engine\response::error('No file was uploaded.', $wrap);
90  }
91  else
92  {
93  //get the path
94  $new_request = new upload(); //$request;
95  //set new request timestamp
96  $new_request->setData([
97  'timestamp' => time(NULL)
98  ]);
99  $new_request->setParameterStatus(array(
100  'final' => array('timestamp'),
101  'missing' => array(),
102  'extra' => array(),
103  'ids' => array()
104  ));
105 
106  //make request
107  $path = $new_request->path($new_request, true);
108 
109  //test existence
110  if ($path['code'] == 200 AND $path['data'] == true)
111  {
112  //use $path['message']
113  $upload_dir = $path['message']['absolute'];
114 
115  //check for title
116  $title = $data['title'];
117 
118  if ($title == '')
119  {
120  $title = $data['file']['name'];
121  }
122 
123  $destination_path = $upload_dir . $title;
124  $file_path = $path['message']['relative'] . $title;
125 
126  //check if overwrite is true
127  $do_overwrite = $data['overwrite'];
128  if (\iriki\engine\type::is_type($do_overwrite, 'boolean'))
129  {
130  $do_overwrite = \iriki\engine\type::ctype($do_overwrite, 'boolean');
131 
132  if (!$do_overwrite)
133  {
134  //check to see it doesn't already exist
135  if (file_exists($destination_path))
136  {
137  return \iriki\engine\response::error("File already exists.", $wrap);
138  }
139  }
140  }
141  else
142  {
143  return \iriki\engine\response::error("Supplied 'overwrite' value is not boolean.", $wrap);
144  }
145 
146  if (move_uploaded_file($temp_path, $destination_path))
147  {
148  //build upload object
149  $upload = array(
150  'title' => $title,
151  'path' => $file_path,
152  'tag' => $data['tag']
153  );
154 
155  $request->setData($upload);
156  $request->setParameterStatus([
157  'final' => array('title', 'path', 'tag'),
158  'missing' => array(),
159  'extra' => array(),
160  'ids' => array()
161  ]);
162 
163  $result = $request->create($request, $wrap);
164 
165  return $result;
166  }
167  else
168  {
169  return \iriki\engine\response::error('Some error occurred while uploading file.', $wrap);
170  }
171  }
172  else
173  {
174  return \iriki\engine\response::error('Some error occurred while preparing upload path.', $wrap);
175  }
176  }
177  }
